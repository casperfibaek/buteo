 253    min_size: int = 1,

 255):
 256    height, width, channels = arr.shape
 257    mask_shape = (height, width, channels)
 258
 259    if not channel_last:
 260        channels, height, width = arr.shape
 261        mask_shape = (channels, height, width)
 262
 263    mask = np.ones(mask_shape, dtype=np.uint8)
 264    zero = np.array(0, dtype=np.uint8)
 265
 266    mask_height = (np.random.uniform(0.0, 1.0, size=(height, )) < p).astype(np.uint8)
 267    mask_width = (np.random.uniform(0.0, 1.0, size=(width, )) < p).astype(np.uint8)
 268
 269    mask_height_indices = np.where(mask_height == 1)[0]
 270    mask_width_indices = np.where(mask_width == 1)[0]
 271
 272    vertical_lines = mask_height.sum()
 273    horizontal_lines = mask_width.sum()
 274
 275    if vertical_lines == 0 and horizontal_lines == 0:
 276        return mask
 277    
 278    vertical_start = np.random.randint(0, height, size=(vertical_lines, ))
 279    vertical_end = vertical_start + np.floor(np.random.uniform(min_height, max_height, size=(vertical_lines, )) * height).astype(np.int64)
 280
 281    for i in range(vertical_end.shape[0]):
 282        if vertical_end[i] > height:
 283            vertical_end[i] = height
 284
 285    horizontal_start = np.random.randint(0, width, size=(horizontal_lines, ))
 286    horizontal_end = horizontal_start + np.floor(np.random.uniform(min_width, max_width, size=(horizontal_lines, )) * width).astype(np.int64)
 287
 288    for j in range(horizontal_end.shape[0]):
 289        if horizontal_end[j] > width:
 290            horizontal_end[j] = width
 291
 292    if channel_last:
 293        for idx, pos_y in enumerate(mask_height_indices):
 294            size = np.random.randint(min_size, max_size + 1)
 295            size_half = size // 2
 296
 297            mask[vertical_start[idx]:vertical_end[idx], pos_y - size_half:pos_y + size_half + 1 , :] = zero
 298
 299        for idx, pos_x in enumerate(mask_width_indices):
 300            size = np.random.randint(min_size, max_size + 1)
 301            size_half = size // 2
 302
 303            mask[pos_x - size_half: pos_x + size_half + 1, horizontal_start[idx]:horizontal_end[idx], :] = zero
 304    else:
 305        for idx, pos_y in enumerate(mask_height_indices):
 306            mask[:, vertical_start[idx]:vertical_end[idx], pos_y + size] = zero
 307
 308        for idx, pos_x in enumerate(mask_width_indices):
 309            mask[:, pos_x + size, horizontal_start[idx]:horizontal_end[idx]] = zero

 407@jit(nopython=True, nogil=True, cache=True, fastmath=True)
 408def _round_converter(idx, height, width):
 409    # Top side
 410    if idx < width:
 411        return 0, idx
 412
 413    # Right side
 414    elif idx < width + height - 1:
 415        return idx - width + 1, width - 1
 416
 417    # Bottom side
 418    elif idx < 2 * width + height - 2:
 419        return height - 1, width - 1 - (idx - width - height + 2)
 420
 421    # Left side
 422    else:
 423        return height - 1 - (idx - 2 * width - height + 3), 0

 426@jit(nopython=True, nogil=True, cache=True, fastmath=True)
 427def _quadratic_bezier(px0, py0, px1, py1, px2, py2, t):
 428    """Calculate point on a quadratic Bezier curve using individual components."""
 429    a = (1-t)
 430    b = t
 431    x = a*(a*px0 + b*px1) + b*(a*px1 + b*px2)
 432    y = a*(a*py0 + b*py1) + b*(a*py1 + b*py2)
 433    
 434    return int(x + 0.5), int(y + 0.5)  # Using +0.5 and int for rounding

 437@jit(nopython=True, nogil=True, cache=True, fastmath=True, parallel=True)
 438def mask_lines_2d_bezier(
 439    arr: np.ndarray,
 440    p: float = 0.05,
 441    channel_last: bool = True,
 442) -> np.ndarray:
 443    if channel_last:
 444        height, width, channels = arr.shape
 445    else:
 446        channels, height, width = arr.shape
 447
 448    mask = np.ones((height, width, channels), dtype=np.uint8)
 449
 450    border_pixels = (2 * (height + width)) - 4
 451    lines_count = (np.random.rand(border_pixels) < p).sum()
 452
 453    if lines_count == 0:
 454        return mask
 455
 456    _sy = np.zeros(lines_count, dtype=np.int64)
 457    _sx = np.zeros(lines_count, dtype=np.int64)
 458    _ey = np.zeros(lines_count, dtype=np.int64)
 459    _ex = np.zeros(lines_count, dtype=np.int64)
 460    _my = np.zeros(lines_count, dtype=np.int64)
 461    _mx = np.zeros(lines_count, dtype=np.int64)
 462
 463    for i in prange(lines_count):
 464        start_idx = np.random.randint(0, border_pixels)
 465        end_idx = np.random.randint(0, border_pixels)
 466
 467        _sy[i], _sx[i] = _round_converter(start_idx, height, width)
 468        _ey[i], _ex[i] = _round_converter(end_idx, height, width)
 469        _my[i] = np.random.randint(0, height)
 470        _mx[i] = np.random.randint(0, width)
 471
 472    for i in range(lines_count):
 473        # 1. Calculate the Bounding Box for Each Curve:
 474        min_x = min(_sx[i], _mx[i], _ex[i])
 475        max_x = max(_sx[i], _mx[i], _ex[i])
 476        min_y = min(_sy[i], _my[i], _ey[i])
 477        max_y = max(_sy[i], _my[i], _ey[i])
 478        
 479        # 2. Determine the `linspace` Resolution for Each Curve:
 480        bounding_box_diagonal = int(np.sqrt((max_x - min_x)**2 + (max_y - min_y)**2))
 481        space = np.linspace(0.0, 1.0, bounding_box_diagonal)
 482    
 483        for t_idx in prange(space.shape[0]):
 484            t = space[t_idx]
 485            
 486            y, x = _quadratic_bezier(_sx[i], _sy[i], _mx[i], _my[i], _ex[i], _ey[i], t)
 487
 488            if channel_last:
 489                mask[y, x, :] = 0
 490            else:
 491                mask[:, y, x] = 0
 492
 493    return mask
