 28@jit(nopython=True, nogil=True, cache=True)
 29def fit_data_to_dtype(
 30    data: np.ndarray,
 31    dtype: np.dtype,
 32) -> np.ndarray:
 33    """
 34    Fit data to a dtype. If the dtype is an integer type, the data will be
 35        rounded to the nearest integer. If the dtype is a float type, the data
 36        will be clipped to the min and max values of the dtype.
 37    
 38    Args:
 39        data (np.ndarray): The data to fit to the dtype.
 40        dtype (np.dtype): The dtype to fit the data to.
 41
 42    Returns:
 43        np.ndarray: The data fitted to the dtype.
 44    """
 45    max_value = np.iinfo(dtype).max
 46    min_value = np.iinfo(dtype).min
 47
 48    if is_integer_dtype(dtype):
 49        return np.rint(
 50            np.clip(data, min_value, max_value)
 51        ).astype(dtype)
 52
 53    return np.clip(
 54        data,
 55        min_value,
 56        max_value,
 57    ).astype(dtype)

 60@jit(nopython=True, nogil=True, cache=True, fastmath=True, parallel=True)
 61def feather_box_2d(
 62    array: np.ndarray,
 63    bbox: Union[List[int], Tuple[int]],
 64    feather_dist: int = 3,
 65) -> np.ndarray:
 66    """
 67    Feather a box into an array (2D). Box should be the original box
 68        buffered by feather_dist.
 69
 70    Args:
 71        array_whole (np.ndarray): The array containing the box.
 72        bbox (Union[List[int], Tuple[int]]): The box.
 73            the bbox should be in the form [x_min, x_max, y_min, y_max].
 74    
 75    Keyword Args:
 76        feather_dist (int=3): The distance to feather the box.
 77
 78    Returns:
 79        Tuple[np.ndarray, np.ndarray]: The featherweights for the array and the bbox.
 80    """
 81    # Get the bbox
 82    x_min, x_max, y_min, y_max = bbox
 83
 84    kernel_size = int((feather_dist * 2) + 1)
 85    n_offsets = kernel_size ** 2
 86    max_dist = (np.sqrt(2) / 2) + feather_dist
 87
 88    feather_offsets = np.zeros((n_offsets, 2), dtype=np.int64)
 89
 90    within_circle = 0
 91    for i in prange(-feather_dist, feather_dist + 1):
 92        for j in prange(-feather_dist, feather_dist + 1):
 93            dist = np.sqrt(i ** 2 + j ** 2)
 94            if dist <= max_dist:
 95                feather_offsets[within_circle][0] = i
 96                feather_offsets[within_circle][1] = j
 97                within_circle += 1
 98
 99    feather_offsets = feather_offsets[:within_circle]
100    n_offsets = feather_offsets.shape[0]
101
102    x_min = max(0, x_min - feather_dist)
103    x_max = min(x_max + 1 + feather_dist, array.shape[1])
104    y_min = max(0, y_min - feather_dist)
105    y_max = min(y_max + 1 + feather_dist, array.shape[0])
106
107    feather_weights_array = np.ones_like(array, dtype=np.float32)
108
109    # Feather the box
110    for y in prange(y_min, y_max + 1):
111        for x in prange(x_min, x_max + 1):
112            total_possible = n_offsets
113            within_bbox = 0
114            for offset in feather_offsets:
115                x_offset, y_offset = offset
116                x_new = x + x_offset
117                y_new = y + y_offset
118
119                if y_new < 0 or y_new >= array.shape[0]:
120                    total_possible -= 1
121                    continue
122
123                if x_new < 0 or x_new >= array.shape[1]:
124                    total_possible -= 1
125                    continue
126
127                if x_new - feather_dist >= x_min and x_new + feather_dist <= x_max and y_new - feather_dist >= y_min and y_new + feather_dist <= y_max:
128                    within_bbox += 1
129
130            weights_box = within_bbox / total_possible
131            feather_weights_array[y, x] = 1 - weights_box
132
133    return feather_weights_array

198def simple_shift_kernel_2d(
199    x_offset: float,
200    y_offset: float,
201) -> Tuple[np.ndarray, np.ndarray]:
202    """
203    Create a 2D shift kernel. Useful for either aligning rasters at the sub-pixel
204    level or for shifting a raster by a whole pixel while keeping the bbox.
205    Can be used to for an augmentation, where channel misalignment is simulated.
206
207    Args:
208        x_offset (float): The x offset.
209        y_offset (float): The y offset.
210
211    Returns:
212        Tuple[np.ndarray, np.ndarray]: The offsets and weights.
213    """
214    if x_offset == 0.0 and y_offset == 0.0:
215        offsets = np.array([[0, 0]], dtype=np.int64)
216        weights = np.array([1.0], dtype=np.float32)
217
218        return offsets, weights
219
220    y0 = [int(np.floor(y_offset)), int(np.ceil(y_offset))] if y_offset != 0 else [0, 0]
221    x0 = [int(np.floor(x_offset)), int(np.ceil(x_offset))] if x_offset != 0 else [0, 0]
222
223    if x_offset == 0.0 or x_offset % 1 == 0.0:
224        offsets = np.zeros((2, 2), dtype=np.int64)
225        weights = np.zeros(2, dtype=np.float32)
226
227        offsets[0] = [int(x_offset) if x_offset % 1 == 0.0 else 0, y0[0]]
228        offsets[1] = [int(x_offset) if x_offset % 1 == 0.0 else 0, y0[1]]
229
230        weights[0] = y_offset - y0[0]
231        weights[1] = 1 - weights[0]
232
233    elif y_offset == 0.0 or y_offset % 1 == 0.0:
234        offsets = np.zeros((2, 2), dtype=np.int64)
235        weights = np.zeros(2, dtype=np.float32)
236
237        offsets[0] = [x0[0], int(y_offset) if y_offset % 1 == 0.0 else 0]
238        offsets[1] = [x0[1], int(y_offset) if y_offset % 1 == 0.0 else 0]
239
240        weights[0] = x_offset - x0[0]
241        weights[1] = 1 - weights[0]
242
243    else:
244        offsets = np.zeros((4, 2), dtype=np.int64)
245        weights = np.zeros(4, dtype=np.float32)
246
247        offsets[0] = [x0[0], y0[0]]
248        offsets[1] = [x0[0], y0[1]]
249        offsets[2] = [x0[1], y0[0]]
250        offsets[3] = [x0[1], y0[1]]
251
252        weights[0] = (1 - (x_offset - offsets[0][0])) * (1 - (y_offset - offsets[0][1]))
253        weights[1] = (1 - (x_offset - offsets[1][0])) * (1 + (y_offset - offsets[1][1]))
254        weights[2] = (1 + (x_offset - offsets[2][0])) * (1 - (y_offset - offsets[2][1]))
255        weights[3] = (1 + (x_offset - offsets[3][0])) * (1 + (y_offset - offsets[3][1]))
256
257    return offsets, weights

261@jit(nopython=True, nogil=True, cache=True, fastmath=True, parallel=True)
262def convolution_simple(
263    array: np.ndarray,
264    offsets: np.ndarray,
265    weights: np.ndarray,
266    intensity: float = 1.0,
267):
268    """
269    Convolve a kernel with an array using a simple method.
270
271    Args:
272        array (np.ndarray): The array to convolve.
273        offsets (np.ndarray): The offsets of the kernel.
274        weights (np.ndarray): The weights of the kernel.
275    
276    Keyword Args:
277        intensity (float=1.0): The intensity of the convolution. If
278            1.0, the convolution is applied as is. If 0.5, the
279            convolution is applied at half intensity.
280
281    Returns:
282        np.ndarray: The convolved array.
283    """
284    result = np.empty_like(array, dtype=np.float32)
285
286    if intensity <= 0.0:
287        return array.astype(np.float32)
288
289    for col in prange(array.shape[0]):
290        for row in prange(array.shape[1]):
291            result_value = 0.0
292            for i in range(offsets.shape[0]):
293                new_col = col + offsets[i, 0]
294                new_row = row + offsets[i, 1]
295
296                if new_col < 0:
297                    new_col = 0
298                elif new_col >= array.shape[0]:
299                    new_col = array.shape[0] - 1
300
301                if new_row < 0:
302                    new_row = 0
303                elif new_row >= array.shape[1]:
304                    new_row = array.shape[1] - 1
305
306                result_value += array[new_col, new_row] * weights[i]
307
308            result[col, row] = result_value
309
310    if intensity < 1.0:
311        result *= intensity
312        array *= (1.0 - intensity)
313        result += array
314
315    return result

269    """
270    Convolve a kernel with an array using a simple method.
271

273        array (np.ndarray): The array to convolve.

275        weights (np.ndarray): The weights of the kernel.
276    
277    Keyword Args:
278        intensity (float=1.0): The intensity of the convolution. If

281
282    Returns:
283        np.ndarray: The convolved array.
284    """
285    result = np.empty_like(array, dtype=np.float32)
286
287    if intensity <= 0.0:
288        return array.astype(np.float32)
289
290    for col in prange(array.shape[0]):
291        for row in prange(array.shape[1]):
292            result_value = 0.0
293            for i in range(offsets.shape[0]):
294                new_col = col + offsets[i, 0]
295                new_row = row + offsets[i, 1]
296
297                if new_col < 0:
298                    new_col = 0
299                elif new_col >= array.shape[0]:
300                    new_col = array.shape[0] - 1
301
302                if new_row < 0:
303                    new_row = 0
304                elif new_row >= array.shape[1]:
305                    new_row = array.shape[1] - 1
306
307                result_value += array[new_col, new_row] * weights[i]
308
309            result[col, row] = result_value
310
311    if intensity < 1.0:
312        result *= intensity
313        array *= (1.0 - intensity)
314        result += array
315
316    return result
