 280def augmentation_noise_uniform(
 281    X: np.ndarray,
 282    max_amount: float = 0.1,
 283    additive: bool = False,
 284    per_channel: bool = True,
 285    channel_last: bool = True,
 286    inplace: bool = False,
 287) -> np.ndarray:
 288    """Adds random noise seperately to each pixel of the image. The noise works
 289    for both channel first and last images. Follows a uniform distribution.
 290    Input should be (height, width, channels) or (channels, height, width).
 291
 292    NOTE: Beware of datatypes. Consider casting to float32 before adding noise.
 293
 294    Parameters
 295    ----------
 296    X : np.ndarray
 297        The image to add noise to.
 298
 299    max_amount : float, optional
 300        The maximum amount of noise to add, sampled uniformly, default: 0.1.
 301
 302    additive : bool, optional
 303        Whether to add or multiply the noise, default: False.
 304
 305    per_channel : bool, optional
 306        Whether to add the same noise to each channel or different noise to each channel (per_channel), default: True.
 307
 308    channel_last : bool, optional
 309        Whether the image is (channels, height, width) or (height, width, channels), ignored for this function.
 310        Kept to keep the same function signature as other augmentations, default: True.
 311
 312    inplace : bool, optional
 313        Whether to perform the rotation in-place, default: False.
 314
 315    Returns
 316    -------
 317    np.ndarray
 318        The image with uniform noise.
 319    """
 320    if not inplace:
 321        X = X.copy()
 322
 323    amount = np.random.rand() * max_amount
 324
 325    if per_channel:
 326        if additive:
 327            random_noise = np.random.uniform(-amount, amount, size=X.shape).astype(X.dtype)
 328            X[:] += random_noise
 329        else:
 330            random_noise = np.random.uniform(1 - amount, 1 + amount, size=X.shape).astype(X.dtype)
 331            X[:] *= random_noise
 332
 333    else:
 334        if channel_last: # hwc
 335            if additive:
 336                random_noise = np.random.uniform(-amount, amount, size=X.shape[:2]).astype(X.dtype)
 337                for i in range(X.shape[2]):
 338                    X[:, :, i] += random_noise
 339            else:
 340                random_noise = np.random.uniform(1 - amount, 1 + amount, size=X.shape[:2]).astype(X.dtype)
 341                for i in range(X.shape[2]):
 342                    X[:, :, i] *= random_noise
 343        else:
 344            if additive:
 345                random_noise = np.random.uniform(-amount, amount, size=X.shape[1:]).astype(X.dtype)
 346                for i in range(X.shape[0]):
 347                    X[i, :, :] += random_noise
 348            else:
 349                random_noise = np.random.uniform(1 - amount, 1 + amount, size=X.shape[1:]).astype(X.dtype)
 350                for i in range(X.shape[0]):
 351                    X[i, :, :] *= random_noise
 352
 353    return X

 570@jit(nopython=True, nogil=True, cache=True, fastmath=True, parallel=True)
 571def augmentation_contrast(
 572    X: np.ndarray,
 573    max_amount: float = 0.1,
 574    channel_last: bool = True,
 575    inplace: bool = False,
 576) -> np.ndarray:
 577    """Changes the contrast of an image by a random amount, seperately for each channel.
 578    Input should be (height, width, channels) or (channels, height, width).
 579
 580    NOTE: Beware of datatypes. Consider casting to float32 before adding noise.
 581
 582    Parameters
 583    ----------
 584    X : np.ndarray
 585        The image to change the contrast of.
 586
 587    max_amount : float, optional
 588        The max amount to change the contrast by, default: 0.1.
 589
 590    channel_last : bool, optional
 591        Whether the image is (channels, height, width) or (height, width, channels), default: True.
 592
 593    inplace : bool, optional
 594        Whether to perform the rotation in-place, default: False.
 595
 596    Returns
 597    -------
 598    np.ndarray
 599        The image with changed contrast.
 600    """
 601    if not inplace:
 602        X = X.copy()
 603
 604    amount = np.random.rand() * max_amount
 605
 606    channels = X.shape[2] if channel_last else X.shape[0]
 607    mean_pixel = np.zeros(channels, dtype=np.float32)
 608
 609    # Numba workaround because numba does not support axis argument in np.mean
 610    if channel_last:
 611        for i in prange(channels):
 612            mean_pixel[i] = np.nanmean(X[:, :, i])
 613    else:
 614        for i in prange(channels):
 615            mean_pixel[i] = np.nanmean(X[i, :, :])
 616
 617    for i in prange(channels):
 618        X[:, :, i] = (X[:, :, i] - mean_pixel[i]) * (1 + amount) + mean_pixel[i]
 619
 620    return X
